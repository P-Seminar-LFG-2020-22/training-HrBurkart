
#Start:

git init
git add .

git config user.email "j.burkart@lfg.muenchen.musin.de"
git config user.name "Burkart"

git commit -m "erster Commit"

git remote add origin https://github.com/P-Seminar-LFG-2020-22/training-HrBurkart.git

git push -u origin master


#Den aktuellen Stand zu Änderungen, Branches, Commits anzeigen:

git status

#Änderungen anzeigen:

git diff

#Änderungen hochladen:

git pull
git add .
git commit -m "zweiter Commit"
git push -u origin master



#Damit erstellst du einen neuen Zweig, in dem z.B. ein Feature entwickelt werden kann:

git branch

#Master und Branch zusammenfügen:

git checkout master
git merge feature-01

#Historie betrachten

git log
